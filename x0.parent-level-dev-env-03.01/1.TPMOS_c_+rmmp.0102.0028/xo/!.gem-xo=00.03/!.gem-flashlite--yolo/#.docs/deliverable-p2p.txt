OP 1: wallet_login
───────────────────
Purpose: Authenticate user and load wallet
Args: username, password

Implementation:
  - Check if pieces/wallets/users/<username>/profile.txt exists
  - If not, create new user profile
  - Hash password (SHA256 with salt) and compare to stored hash
  - On success:
    * Write wallet address to pieces/wallets/sessions/<username>/active_wallet.txt
    * Write session token to pieces/wallets/sessions/<username>/session.txt
    * Output: "Wallet logged in: <wallet_address>"
  - On failure:
    * Output: "Authentication failed"
    * Return 1

File structure:
  pieces/wallets/
  ├── users/
  │   └── <username>/
  │       ├── profile.txt       # username, wallet_address, created_at
  │       └── password_hash.txt # salt + SHA256 hash
  └── sessions/
      └── <username>/
          ├── active_wallet.txt # current wallet address
          └── session.txt       # session token, expires_at

Test command:
  OP p2p::wallet_login "player1" "mypass123"

OP 2: wallet_create
────────────────────
Purpose: Create additional wallet for authenticated user
Args: username, wallet_name (e.g., "main", "trading", "mining")

Implementation:
  - Verify user is logged in (check session.txt)
  - Generate unique wallet address (SHA256 of username + wallet_name + timestamp)
  - Create wallet directory: pieces/wallets/users/<username>/wallets/<wallet_name>/
  - Initialize balance.txt with 0
  - Output: "Wallet created: <wallet_address>"

File structure:
  pieces/wallets/users/<username>/wallets/
  └── <wallet_name>/
      ├── wallet.txt        # wallet_address, name, created_at
      └── balance.txt       # balance=0

Test command:
  OP p2p::wallet_create "player1" "trading"

OP 3: list_wallets
───────────────────
Purpose: List all wallets for authenticated user
Args: username

Implementation:
  - Verify user is logged in
  - Scan pieces/wallets/users/<username>/wallets/ directory
  - For each wallet, read balance.txt and wallet.txt
  - Output formatted list:
    ```
    Wallets for player1:
    [1] main (abc123...) - Balance: 150.00 TPM
    [2] trading (def456...) - Balance: 50.00 TPM
    ```
  - Write to pieces/wallets/users/<username>/wallet_list.txt for GUI

Test command:
  OP p2p::list_wallets "player1"

OP 4: mine_block
─────────────────
Purpose: Mine coins via SHA256 proof-of-work
Args: username, wallet_address

Implementation:
  - Verify user is logged in
  - Read difficulty from pieces/blockchain/config.txt (default: 2)
  - Create block header: prev_hash + wallet_address + timestamp + nonce
  - Increment nonce until SHA256 hash starts with "00" (difficulty 2)
  - On success:
    * Credit mining reward (10 coins) to wallet balance
    * Append block to pieces/blockchain/chain.txt
    * Output: "Block mined! Reward: 10 TPM (nonce: 12345)"
  - Difficulty scaling:
    * Every 10 blocks, increase difficulty by 1 (max 4)
    * Store difficulty in pieces/blockchain/config.txt

File structure:
  pieces/blockchain/
  ├── config.txt          # difficulty=2, total_blocks=5
  ├── chain.txt           # Block 1: hash, nonce, timestamp, reward
  └── mining_log.txt      # Mining history

Test command:
  OP p2p::mine_block "player1" "abc123..."

OP 5: get_balance
──────────────────
Purpose: Query wallet balance
Args: username, wallet_address

Implementation:
  - Verify user owns wallet (check session)
  - Read pieces/wallets/users/<username>/wallets/<wallet_name>/balance.txt
  - Calculate balance from blockchain transactions (sum all credits - debits)
  - Output: "Balance: 150.00 TPM"
  - Write to pieces/wallets/users/<username>/wallets/<wallet_name>/balance_display.txt

Test command:
  OP p2p::get_balance "player1" "abc123..."

OP 6: send_coins
─────────────────
Purpose: Transfer coins between wallets
Args: from_wallet, to_wallet_address, amount

Implementation:
  - Verify user owns from_wallet (check session)
  - Check from_wallet balance >= amount
  - Deduct amount from sender balance
  - Add amount to receiver balance (find wallet by address)
  - Create transaction record: pieces/blockchain/transactions/tx_<id>.txt
  - Append to chain.txt
  - Output: "Sent 50 TPM to def456..."

File structure:
  pieces/blockchain/transactions/
  └── tx_<timestamp>.txt
      # from=<wallet>
      # to=<wallet>
      # amount=50
      # timestamp=...
      # hash=<SHA256>

Test command:
  OP p2p::send_coins "abc123..." "def456..." 50

/* Read wallet address from active session */
static void read_wallet_address(char *buf, size_t size) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/pieces/wallets/sessions/%s/active_wallet.txt",
             project_root, current_wallet);
    
    FILE *f = fopen(path, "r");
    if (f) {
        fgets(buf, size, f);
        char *nl = strchr(buf, '\n');
        if (nl) *nl = '\0';
        fclose(f);
    } else {
        strncpy(buf, "Not logged in", size - 1);
    }
}

/* Read subnet list from network config */
static void read_subnet_list(char *buf, size_t size) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/pieces/network/subnets",
             project_root);
    
    buf[0] = '\0';
    DIR *dir = opendir(path);
    if (dir) {
        struct dirent *entry;
        int first = 1;
        while ((entry = readdir(dir)) != NULL) {
            if (entry->d_name[0] == '.') continue;
            if (!first) strncat(buf, ", ", size - strlen(buf) - 1);
            strncat(buf, entry->d_name, size - strlen(buf) - 1);
            first = 0;
        }
        closedir(dir);
    }
    if (strlen(buf) == 0) strncpy(buf, "None", size - 1);
}

/* Read inbox counts */
static void read_inbox_counts(int *unread, int *total) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/pieces/chat/inbox/%s/status.txt",
             project_root, current_wallet);
    
    *unread = 0;
    *total = 0;
    
    FILE *f = fopen(path, "r");
    if (f) {
        char line[MAX_LINE];
        while (fgets(line, sizeof(line), f)) {
            if (strncmp(line, "unread_count=", 13) == 0) {
                *unread = atoi(line + 13);
            } else if (strncmp(line, "total_messages=", 15) == 0) {
                *total = atoi(line + 15);
            }
        }
        fclose(f);
    }
}

/* Read inbox list for GUI (first 10 messages) */
static void read_inbox_list(char *buf, size_t size) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/pieces/chat/inbox/%s/inbox_list.txt",
             project_root, current_wallet);
    
    buf[0] = '\0';
    FILE *f = fopen(path, "r");
    if (f) {
        fread(buf, 1, size - 1, f);
        fclose(f);
    } else {
        /* Fallback: generate from msg files */
        char msg_dir[MAX_PATH];
        snprintf(msg_dir, sizeof(msg_dir),
                 "%s/projects/p2p-net/pieces/chat/inbox/%s",
                 project_root, current_wallet);
        
        DIR *dir = opendir(msg_dir);
        if (dir) {
            struct dirent *entry;
            int count = 0;
            while ((entry = readdir(dir)) != NULL && count < 10) {
                if (strncmp(entry->d_name, "msg_", 4) != 0) continue;
                
                /* Read message subject */
                char msg_path[MAX_PATH];
                snprintf(msg_path, sizeof(msg_path),
                         "%s/%s", msg_dir, entry->d_name);
                
                FILE *mf = fopen(msg_path, "r");
                if (mf) {
                    char line[MAX_LINE];
                    char subject[128] = "";
                    char from[64] = "";
                    int is_read = 0;
                    
                    while (fgets(line, sizeof(line), mf)) {
                        if (strncmp(line, "subject=", 8) == 0) {
                            char *nl = strchr(line + 8, '\n');
                            if (nl) *nl = '\0';
                            strncpy(subject, line + 8, sizeof(subject) - 1);
                        } else if (strncmp(line, "from=", 5) == 0) {
                            char *nl = strchr(line + 5, '\n');
                            if (nl) *nl = '\0';
                            strncpy(from, line + 5, sizeof(from) - 1);
                        } else if (strncmp(line, "read=", 5) == 0) {
                            is_read = atoi(line + 5);
                        }
                    }
                    fclose(mf);
                    
                    char icon = is_read ? ' ' : '!';
                    char msg_line[256];
                    snprintf(msg_line, sizeof(msg_line),
                             "[%c] %s - %s\n", icon, from, subject);
                    strncat(buf, msg_line, size - strlen(buf) - 1);
                    count++;
                }
            }
            closedir(dir);
        }
        if (strlen(buf) == 0) strncpy(buf, "No messages", size - 1);
    }
}

/* Read peer count and list */
static void read_peer_info(int *count, char *list_buf, size_t list_size) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/pieces/network/peer_list.txt",
             project_root);
    
    *count = 0;
    list_buf[0] = '\0';
    
    FILE *f = fopen(path, "r");
    if (f) {
        /* First line is count */
        char line[MAX_LINE];
        if (fgets(line, sizeof(line), f)) {
            *count = atoi(line);
        }
        
        /* Rest is formatted list */
        fread(list_buf, 1, list_size - 1, f);
        fclose(f);
    } else {
        /* Fallback: read from peers.txt */
        char peers_path[MAX_PATH];
        snprintf(peers_path, sizeof(peers_path),
                 "%s/projects/p2p-net/pieces/network/peers.txt",
                 project_root);
        
        f = fopen(peers_path, "r");
        if (f) {
            char line[MAX_LINE];
            while (fgets(line, sizeof(line), f)) {
                if (line[0] == '#') continue;
                
                char ip[64], port[16], wallet[64], subnet[32], status[32];
                if (sscanf(line, "%[^|]|%[^|]|%[^|]|%[^|]|%s",
                           ip, port, wallet, subnet, status) == 5) {
                    char peer_line[256];
                    snprintf(peer_line, sizeof(peer_line),
                             "%s:%s (%s) [%s] - %s\n",
                             ip, port, wallet, subnet, status);
                    strncat(list_buf, peer_line, list_size - strlen(list_buf) - 1);
                    (*count)++;
                }
            }
            fclose(f);
        }
        if (strlen(list_buf) == 0) strncpy(list_buf, "No peers", list_size - 1);
    }
}

/* Read message preview (current message) */
static void read_message_preview(char *buf, size_t size) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/pieces/chat/inbox/%s/current_message.txt",
             project_root, current_wallet);
    
    buf[0] = '\0';
    FILE *f = fopen(path, "r");
    if (f) {
        fread(buf, 1, size - 1, f);
        fclose(f);
    } else {
        strncpy(buf, "Select a message to preview", size - 1);
    }
}

static void write_gui_state() {
    char path[MAX_PATH];
    snprintf(path, sizeof(path),
             "%s/projects/p2p-net/manager/gui_state.txt",
             project_root);
    
    FILE *f = fopen(path, "w");
    if (!f) return;
    
    /* Read all current values */
    char wallet_addr[128] = "";
    char subnet_list[512] = "";
    char inbox_list[2048] = "";
    char peer_list[1024] = "";
    char msg_preview[512] = "";
    int unread = 0, total = 0, peer_count = 0;
    
    read_wallet_address(wallet_addr, sizeof(wallet_addr));
    read_subnet_list(subnet_list, sizeof(subnet_list));
    read_inbox_counts(&unread, &total);
    read_inbox_list(inbox_list, sizeof(inbox_list));
    read_peer_info(&peer_count, peer_list, sizeof(peer_list));
    read_message_preview(msg_preview, sizeof(msg_preview));
    
    /* Write in format parser can substitute */
    fprintf(f, "wallet_address=%s\n", wallet_addr);
    fprintf(f, "subnet_list=%s\n", subnet_list);
    fprintf(f, "unread_count=%d\n", unread);
    fprintf(f, "total_count=%d\n", total);
    fprintf(f, "inbox_list=%s\n", inbox_list);
    fprintf(f, "message_preview=%s\n", msg_preview);
    fprintf(f, "peer_count=%d\n", peer_count);
    fprintf(f, "peer_list=%s\n", peer_list);
    fprintf(f, "p2p_response=%s\n", last_response);
    fprintf(f, "last_key=%d\n", last_key_pressed);
    
    fclose(f);
}

        set_response("Left subnet");
        
        /* Refresh peer list */
        write_gui_state();
    }

    write_state();
    write_gui_state();  /* Always update GUI state after key press */
}

================================================================================
OP IMPLEMENTATION PATTERN (Follow This Template)
================================================================================

All ops should follow this pattern (from compose_message.c):

1. **Header**: Purpose, args, GUI integration notes
2. **Includes**: stdio, stdlib, string, unistd, time, limits
3. **Helper functions**: 
   - get_base_path() or resolve project root
   - ensure_dir() for directory creation
   - read/write state files
4. **main()**:
   - Parse and validate arguments
   - Get base path
   - Create directories if needed
   - Perform operation
   - Update state files
   - Output result (printf for logging)
   - Return 0 on success, 1 on failure

CPU SAFETY:
- Ops are short-lived (fork/exec/exit pattern)
- No long-running loops or daemons
- Use system() for simple mkdir (acceptable in ops)
- For network ops, use fork/exec for external commands

EXAMPLE: wallet_login.c Structure
```c
/*
 * wallet_login.c - P2P-NET Wallet Authentication Op
 * Purpose: Authenticate user and load wallet
 * Args: username, password
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <time.h>
#include <limits.h>

#define MAX_PATH PATH_MAX
#define MAX_LINE 1024

static void get_base_path(char *buf, size_t size) {
    if (getcwd(buf, size) != NULL) return;
    strncpy(buf, ".", size - 1);
}

static void ensure_dir(const char *path) {
    char cmd[MAX_PATH * 2];
    snprintf(cmd, sizeof(cmd), "mkdir -p %s", path);
    system(cmd);
}

/* SHA256 hash function - use openssl or implement */
static void sha256_hash(const char *input, char *output) {
    /* Use: echo -n "input" | sha256sum | awk '{print $1}' */
    char cmd[MAX_PATH * 2];
    snprintf(cmd, sizeof(cmd),
             "echo -n '%s' | sha256sum | awk '{print $1}'", input);
    
    FILE *f = popen(cmd, "r");
    if (f) {
        fgets(output, 65, f);
        pclose(f);
    }
}

int main(int argc, char *argv[]) {
    char base_path[MAX_PATH];
    
    /* Parse arguments */
    if (argc < 3) {
        fprintf(stderr, "Usage: wallet_login <username> <password>\n");
        return 1;
    }
    
    const char *username = argv[1];
    const char *password = argv[2];
    
    /* Get base path */
    get_base_path(base_path, sizeof(base_path));
    
    /* Check if user exists */
    char user_path[MAX_PATH];
    snprintf(user_path, sizeof(user_path),
             "%s/projects/p2p-net/pieces/wallets/users/%s",
             base_path, username);
    
    if (access(user_path, F_OK) == -1) {
        /* Create new user */
        printf("Creating new user: %s\n", username);
        
        char profile_path[MAX_PATH];
        snprintf(profile_path, sizeof(profile_path),
                 "%s/projects/p2p-net/pieces/wallets/users/%s",
                 base_path, username);
        ensure_dir(profile_path);
        
        /* Generate wallet address */
        char wallet_addr[65];
        char seed[MAX_LINE];
        snprintf(seed, sizeof(seed), "%s_%ld", username, time(NULL));
        sha256_hash(seed, wallet_addr);
        
        /* Write profile */
        char prof_file[MAX_PATH];
        snprintf(prof_file, sizeof(prof_file),
                 "%s/profile.txt", profile_path);
        FILE *f = fopen(prof_file, "w");
        if (f) {
            fprintf(f, "username=%s\n", username);
            fprintf(f, "wallet_address=%s\n", wallet_addr);
            fprintf(f, "created_at=%ld\n", time(NULL));
            fclose(f);
        }
        
        /* Create main wallet */
        char wallet_path[MAX_PATH];
        snprintf(wallet_path, sizeof(wallet_path),
                 "%s/wallets/main", profile_path);
        ensure_dir(wallet_path);
        
        char wallet_file[MAX_PATH];
        snprintf(wallet_file, sizeof(wallet_file),
                 "%s/wallet.txt", wallet_path);
        f = fopen(wallet_file, "w");
        if (f) {
            fprintf(f, "wallet_address=%s\n", wallet_addr);
            fprintf(f, "name=main\n");
            fprintf(f, "created_at=%ld\n", time(NULL));
            fclose(f);
        }
        
        /* Initialize balance */
        char bal_file[MAX_PATH];
        snprintf(bal_file, sizeof(bal_file),
                 "%s/balance.txt", wallet_path);
        f = fopen(bal_file, "w");
        if (f) {
            fprintf(f, "balance=0\n");
            fclose(f);
        }
    }
    
    /* Hash password and store (or verify) */
    char hash_input[MAX_LINE];
    snprintf(hash_input, sizeof(hash_input), "%s_%s", username, password);
    char hash[65];
    sha256_hash(hash_input, hash);
    
    char hash_path[MAX_PATH];
    snprintf(hash_path, sizeof(hash_path),
             "%s/projects/p2p-net/pieces/wallets/users/%s/password_hash.txt",
             base_path, username);
    
    /* Check if password matches or create new */
    FILE *f = fopen(hash_path, "r");
    if (f) {
        char stored_hash[65];
        fgets(stored_hash, sizeof(stored_hash), f);
        fclose(f);
        
        if (strcmp(hash, stored_hash) != 0) {
            fprintf(stderr, "Authentication failed\n");
            return 1;
        }
    } else {
        /* Store new password hash */
        ensure_dir(hash_path);  /* Creates parent dirs */
        f = fopen(hash_path, "w");
        if (f) {
            fprintf(f, "%s\n", hash);
            fclose(f);
        }
    }
    
    /* Create session */
    char session_path[MAX_PATH];
    snprintf(session_path, sizeof(session_path),
             "%s/projects/p2p-net/pieces/wallets/sessions/%s",
             base_path, username);
    ensure_dir(session_path);
    
    /* Write active wallet */
    char active_file[MAX_PATH];
    snprintf(active_file, sizeof(active_file),
             "%s/active_wallet.txt", session_path);
    f = fopen(active_file, "w");
    if (f) {
        /* Read wallet address from profile */
        char prof_file[MAX_PATH];
        snprintf(prof_file, sizeof(prof_file),
                 "%s/projects/p2p-net/pieces/wallets/users/%s/profile.txt",
                 base_path, username);
        
        char wallet_addr[65] = "";
        FILE *pf = fopen(prof_file, "r");
        if (pf) {
            char line[MAX_LINE];
            while (fgets(line, sizeof(line), pf)) {
                if (strncmp(line, "wallet_address=", 15) == 0) {
                    char *nl = strchr(line + 15, '\n');
                    if (nl) *nl = '\0';
                    strncpy(wallet_addr, line + 15, sizeof(wallet_addr) - 1);
                }
            }
            fclose(pf);
        }
        
        fprintf(f, "%s\n", wallet_addr);
        fclose(f);
    }
    
    /* Write session */
    char sess_file[MAX_PATH];
    snprintf(sess_file, sizeof(sess_file),
             "%s/session.txt", session_path);
    f = fopen(sess_file, "w");
    if (f) {
        fprintf(f, "username=%s\n", username);
        fprintf(f, "logged_in=%ld\n", time(NULL));
        fprintf(f, "expires=%ld\n", time(NULL) + 3600);  /* 1 hour */
        fclose(f);
    }
    
    /* Output */
    printf("Wallet logged in: %s\n", username);
    printf("Wallet address: %s\n", wallet_addr);
    
    return 0;
}
```
