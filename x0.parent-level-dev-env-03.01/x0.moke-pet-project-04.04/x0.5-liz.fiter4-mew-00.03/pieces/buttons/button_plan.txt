## BUTTON BEHAVIOR
================================================================================

### Root Level Launchers

#### button.ps1 (Windows)
```powershell
# Usage: .\button.ps1 <action>
# Actions: compile, run, kill, debug, watchdog, deps

param([string]$action = "help")

$SCRIPT_DIR = Split-Path -Parent $MyInvocation.MyCommand.Path
$BUTTON_DIR = Join-Path $SCRIPT_DIR "#.buttons\windows"

switch ($action) {
    "compile" { & "$BUTTON_DIR\compile.ps1" }
    "run"     { & "$BUTTON_DIR\run.ps1" }
    "kill"    { & "$BUTTON_DIR\kill.ps1" }
    "debug"   { & "$BUTTON_DIR\debug_gl.ps1" }
    "watchdog"{ & "$BUTTON_DIR\watchdog.ps1" }
    "deps"    { & "$BUTTON_DIR\deps.ps1" }
    "help"    {
        Write-Host "Usage: .\button.ps1 <action>" -ForegroundColor Cyan
        Write-Host "Actions: compile, run, kill, debug, watchdog, deps"
    }
    default   { Write-Error "Unknown action: $action" }
}
```

#### button.sh (Linux/macOS)
```bash
#!/bin/bash
# Usage: ./button.sh <action>
# Actions: compile, run, kill, debug, watchdog

ACTION="${1:-help}"
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
BUTTON_DIR="$SCRIPT_DIR/#.buttons/linux"

case "$ACTION" in
    compile) bash "$BUTTON_DIR/compile.sh" ;;
    run)     bash "$BUTTON_DIR/run.sh" ;;
    kill)    bash "$BUTTON_DIR/kill.sh" ;;
    debug)   bash "$BUTTON_DIR/debug_gl.sh" ;;
    watchdog) bash "$BUTTON_DIR/watchdog.sh" ;;
    help)
        echo "Usage: ./button.sh <action>"
        echo "Actions: compile, run, kill, debug, watchdog"
        ;;
    *) echo "Unknown action: $ACTION" ;;
esac
```
